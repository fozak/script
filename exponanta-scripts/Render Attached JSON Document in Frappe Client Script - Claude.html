    javascript
    frappe.ui.form.on('Task', {
    refresh: function(frm) {
        // Add the "Edit Document" button to the form
        frm.add_custom_button(__('Edit Document'), function() {
            // Get attachments for the current document
            frappe.call({
                method: 'frappe.client.get_value',
                args: {
                    doctype: 'File',
                    filters: {
                        'attached_to_doctype': frm.doctype,
                        'attached_to_name': frm.docname,
                        'file_name': ['like', '%.json']
                    },
                    fieldname: ['name', 'file_url']
                },
                callback: function(response) {
                    if (response.message) {
                        const file = response.message;
                        
                        // Fetch the JSON file content
                        fetch(file.file_url)
                            .then(response => response.json())
                            .then(jsonData => {
                                // Check if the JSON contains doctype information
                                if (!jsonData.doctype) {
                                    frappe.throw(__('Invalid JSON format: Missing doctype information'));
                                    return;
                                }
                                
                                // Create a new temporary document to render the JSON data
                                const doctype = jsonData.doctype;
                                delete jsonData.doctype; // Remove doctype from the data object
                                
                                // Use a dialog to render the form
                                const d = new frappe.ui.Dialog({
                                    title: __('Edit Document'),
                                    fields: [
                                        {
                                            fieldtype: 'HTML',
                                            fieldname: 'document_html'
                                        }
                                    ],
                                    primary_action_label: __('Save'),
                                    primary_action: function() {
                                        const tempForm = d.fields_dict.document_html.wrapper.querySelector('form');
                                        const formData = new FormData(tempForm);
                                        
                                        // Convert form data to JSON
                                        const updatedData = {};
                                        for (let [key, value] of formData.entries()) {
                                            updatedData[key] = value;
                                        }
                                        updatedData.doctype = doctype;
                                        
                                        // Save the updated JSON back to the attachment
                                        saveJsonAttachment(frm, file.name, updatedData);
                                        
                                        d.hide();
                                    }
                                });
                                
                                d.show();
                                
                                // Create a temporary form to render the document
                                frappe.model.with_doctype(doctype, function() {
                                    const tempForm = frappe.model.get_new_doc(doctype);
                                    
                                    // Apply JSON data to the temporary form
                                    Object.keys(jsonData).forEach(field => {
                                        tempForm[field] = jsonData[field];
                                    });
                                    
                                    // Create a new form for the dialog
                                    const formView = new frappe.ui.form.Form(
                                        doctype,
                                        d.fields_dict.document_html.wrapper,
                                        tempForm
                                    );
                                    
                                    formView.refresh();
                                });
                            })
                            .catch(error => {
                                frappe.throw(__('Error loading JSON file: ') + error.message);
                            });
                    } else {
                        frappe.msgprint(__('No JSON attachments found for this document.'));
                    }
                }
            });
        });
        
        // Function to save the JSON back to an attachment
        function saveJsonAttachment(frm, file_name, jsonData) {
            const jsonString = JSON.stringify(jsonData, null, 2);
            const blob = new Blob([jsonString], { type: 'application/json' });
            
            // Create FormData to upload the file
            const formData = new FormData();
            formData.append('file', blob, 'document.json');
            formData.append('doctype', frm.doctype);
            formData.append('docname', frm.docname);
            formData.append('is_private', 1);
            
            // First delete the existing file
            frappe.call({
                method: 'frappe.client.delete',
                args: {
                    doctype: 'File',
                    name: file_name
                },
                callback: function() {
                    // Upload the new file
                    $.ajax({
                        url: '/api/method/upload_file',
                        type: 'POST',
                        data: formData,
                        processData: false,
                        contentType: false,
                        success: function(response) {
                            frappe.show_alert({
                                message: __('Document updated successfully'),
                                indicator: 'green'
                            });
                            frm.reload_doc();
                        },
                        error: function(error) {
                            frappe.throw(__('Error saving document: ') + error.message);
                        }
                    });
                }
            });
        }
    }
});

    There are a few ways to handle custom fields:

    Option 1: Create Actual Custom Fields in the System

    The best approach would be to actually create these custom fields in your Frappe system:

    1. Go to Setup > Customize > Customize Form
    2. Select "Customer" as the form
    3. Add your custom fields (loyalty_points, account_manager, contract_renewal_date) with appropriate field types
    4. Save the changes

    Once you do this, these fields will appear naturally in the form when you use the Edit Document feature.
