CW._config.columnTypes   — column actions

CW._executeAction becomes the universal dispatcher:
javascriptCW._executeAction = function(action, run_doc, record) {
  if (action.signal) {
    // FSM signal
    return run_doc.child({
      operation:      'update',
      target_doctype: run_doc.target_doctype,
      query:          { where: { name: record.name } },
      input:          { _state: { [action.signal]: '' } },
      options:        { render: false, internal: true },
    })
  }
  // standard run
  return run_doc.child({
    ...action,
    target_doctype: action.target_doctype || run_doc.target_doctype,
    query:          { where: { name: record?.name } },
  })
}
UI components then have zero CW awareness:
javascript// column cell — pure UI, no CW imports

The UI becomes a rendering surface only. All execution logic is in the registry. All behavior is declared in schema/config. The component has no opinion about what clicking does — it just calls CW._executeAction with whatever descriptor it was given.
